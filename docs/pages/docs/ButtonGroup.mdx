import ButtonGroup from '../../../dist/ButtonGroup';
import Button from '../../../dist/Button';

### Description

Button Group gives you the ability to manage lot of buttons in a group

### Available Props

- Required
  - gap &rarr; `string`

### Usage

```js
import { ButtonGroup } from 'recompo';
// OR
import ButtonGroup from 'recompo/ButtonGroup';

return (
  <>
    <ButtonGroup gap={6}>
      <Button variant='primary'>primary</Button>
      <Button variant='secondary'>secondary</Button>
      <Button variant='success'>success</Button>
    </ButtonGroup>
  </>
);
```
