import Button from "../../../dist/Button"; 

### Description

A Button Component with different variants

### Available Props
- Optional
   - variant &rarr; `primary` `success` `error` `info` `warning` `inherit` `secondary`
   - disabled &rarr; If set to `true` button will be disabled. Defaults `false`

### Usage

```js
import { Button } from "recompo";
// OR
import Button "recompo/Button";

const App = () => {
  return (
    <>
      <Button variant="primary">Hello World</Button>
    </>
  );
};
```
